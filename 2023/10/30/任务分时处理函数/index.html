    耗时任务分片处理

    问题模拟

    在前端开发中时常有大量任务处理，导致浏览器主线程长时间忙于处理这些操作，没有时间渲染页面，页面卡顿。

    <button id="bun">dianji</button>

<script>
 const but = document.getElementById('bun')  //获取按钮
    const datas = Array.from({length: 100000}, (_, index) => index)//创建数组
    but.onclick = () => {
        const div = document.createElement('div')
        div.innerText = i
        document.body.appendChild(div)
    }   // 创建100000个元素放在页面
</script>

    可以看到在页面中有明显卡顿，然后元素创建完毕，页面继续渲染。

    问题处理

    创建函数处理任务，把大量任务分块处理，给浏览器渲染页面的时间

    关键位置处理流程：

    1. 使用函数获取当前到下一帧渲染渲染所剩余的时间
    2. 判断剩余时间大于零，并且任务长度大于任务进度值，循环依次执行任务列表中的任务。
    3. 如果剩余时间不大于零，跳出循环递归调用自己。把需要执行的任务放在requestIdleCallback的异步回调中。等待执行时机
    <button id="bun">dianji</button>

<script>
 const but = document.getElementById('bun')  //获取按钮
    const datas = Array.from({length: 100000}, (_, index) => index)//创建数组
    but.onclick = () => {
        chunked(datas)
    }   // 创建100000个元素放在页面
    
    function chunked(datas) {
        if (datas.length === 0) return  //判断任务长度，边界情况
        let index = 0     // 储存当前任务进行到的位置

        function tesk() {
            if (datas.length < index) return   // 如果任务的长度小于任务的进度代表任务完成，结束任务
            requestIdleCallback((res) => {  // (1)
                while (res.timeRemaining() > 0 && datas.length > index) {  //(2)
                    let div = document.createElement('div')
                    div.innerText = datas[index]
                    document.body.appendChild(div)
                    index++
                }
                tesk() //(3)
            })
        }

        tesk()
    }
</script>

    优化

    在代码编写过程中可能存在多个此类情况，上面代码中执行任务流程写入了任务分片函数。提取任务流程。使任务分片函数成为通用函数

    <button id="bun">dianji</button>

<script>
 const but = document.getElementById('bun')  //获取按钮
    const datas = Array.from({length: 100000}, (_, index) => index)//创建数组
    but.onclick = () => {
       let fun = (data, index) => {
            let div = document.createElement('div')
            div.innerText = data
            document.body.appendChild(div)
        }  // 提取任务执行流程
        chunked(datas, fun)
    }   // 创建100000个元素放在页面
    
    function chunked(datas, fun) {
        if (datas.length === 0) return  //判断任务长度，边界情况
        let index = 0     // 储存当前任务进行到的位置

        function tesk() {
            if (datas.length < index) return   // 如果任务的长度小于任务的进度代表任务完成，结束任务
            requestIdleCallback((res) => {
                while (res.timeRemaining() > 0 && datas.length > index) {
                    fun(datas[index], index) // 执行任务流程
                    index++
                }
                tesk()
            })
        }

        tesk()
    }
</script>

    再次优化

    上面优化过程默认使用了requestIdleCallback函数为函数调度器, 再次优化把调度器选择，和函数间隔时间的选择权交给函数使用者

    1. 添加第三个参数调度器
    2. 自定义使用者自己定义调度器的函数，可以使用requestIdleCallback，也可以使用计时函数指定任务运行多少毫秒
    
<button id="bun">dianji</button>

<script>
 const but = document.getElementById('bun')  //获取按钮
    const datas = Array.from({length: 100000}, (_, index) => index)//创建数组
    but.onclick = () => {
       let fun = (data, index) => {
            let div = document.createElement('div')
            div.innerText = data
            document.body.appendChild(div)
        }  // 提取任务执行流程
       
        let diaoduqi = (fun) => {  //(2)
            requestIdleCallback((res) => {
                fun(
                    () => res.timeRemaining() > 0  // 判断当前任务是否超时，把任务调用时机交给开发人员
                )
            })

        }
        chunked(datas, fun,diaoduqi) //（1）
    }   // 创建100000个元素放在页面
    
    function chunked(datas, fun) {
        if (datas.length === 0) return  //判断任务长度，边界情况
        let index = 0     // 储存当前任务进行到的位置

        function tesk() {
            if (datas.length < index) return   // 如果任务的长度小于任务的进度代表任务完成，结束任务
            requestIdleCallback((res) => {
                while (res.timeRemaining() > 0 && datas.length > index) {
                    fun(datas[index], index) // 执行任务流程
                    index++
                }
                tesk()
            })
        }

        tesk()
    }
</script>
